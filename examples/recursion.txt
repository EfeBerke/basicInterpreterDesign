(* recursion.txt - fibonacci *)

let fibonacci = fun(n) ->
    if n <= 1 then
        n
    else
        fibonacci(n - 1) + fibonacci(n - 2)
    end
end;

print(fibonacci(0));
print(fibonacci(1));
print(fibonacci(2));
print(fibonacci(3));
print(fibonacci(4));
print(fibonacci(5));
print(fibonacci(6));
print(fibonacci(7));
